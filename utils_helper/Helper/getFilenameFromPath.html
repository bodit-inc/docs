static getFilenameFromPath(String path) {
  RegExp regExp = RegExp(r'"(.*?)\"', caseSensitive: false, multiLine: false);
  String filename = basename(regExp.stringMatch(path).toString().replaceAll('"', ''));
  return filename;
}
