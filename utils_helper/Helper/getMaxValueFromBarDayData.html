double getMaxValueFromBarDayData({required List<BarDayData> barDayDatas}) {
  List<double> tmp = [];
  for (int i = 0; i < barDayDatas.length; i++) {
    for (int j = 0; j < barDayDatas[i].barHourData.length; j++) {
      if (barDayDatas[i].barHourData[j] == null) continue;
      tmp.add(barDayDatas[i].barHourData[j]!.value);
      tmp.add(barDayDatas[i].barHourData[j]!.realValue);
      tmp.add(barDayDatas[i].barHourData[j]!.realValueBottom);
      tmp.add(barDayDatas[i].barHourData[j]!.groupAvgValue);
      tmp.add(barDayDatas[i].barHourData[j]!.groupRangeTopValue);
      tmp.add(barDayDatas[i].barHourData[j]!.groupRangeTopValueBottom);
      tmp.add(barDayDatas[i].barHourData[j]!.groupRangeBottomValue);
      tmp.add(barDayDatas[i].barHourData[j]!.groupRangeBottomValueBottom);
      tmp.add(barDayDatas[i].barHourData[j]!.totalAvgValue);
      tmp.add(barDayDatas[i].barHourData[j]!.totalRangeTopValue);
      tmp.add(barDayDatas[i].barHourData[j]!.totalRangeTopValueBottom);
      tmp.add(barDayDatas[i].barHourData[j]!.totalRangeBottomValue);
      tmp.add(barDayDatas[i].barHourData[j]!.totalRangeBottomValueBottom);
    }
  }
  return tmp.reduce((curr, next) => curr > next ? curr : next);
}
