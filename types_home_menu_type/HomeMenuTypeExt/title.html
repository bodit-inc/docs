String get title {
  switch (this) {
    case HomeMenuType.healthDoubt:
      return ReportGroup.healthDoubt.name;
    case HomeMenuType.oddAction:
      return ReportGroup.oddAction.name;
    case HomeMenuType.needCheck:
      return ReportGroup.needCheck.name;
    case HomeMenuType.question:
      return "질문하기";
    default:
      return "";
  }
}
