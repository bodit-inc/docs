Widget DropDownMenu({List<dynamic> values = const [], dynamic value, Function? onChanged, String hint = "선택하세요"}) {
  return Container(
    padding: EdgeInsets.only(left: 12),
    decoration: BoxDecoration(
      border: Border.all(color: CustomColors.white_five, width: 1),
    ),
    child: DropdownButtonHideUnderline(
      child: DropdownButton(
        value: value,
        hint: Text(hint, style: TextStyle(fontSize: 13)),
        items: values.map(
          (e) {
            String _text = "전체";
            if (e is CowGroup) {
              _text = e.name;
            } else if (e is Cow) {
              _text = "${e.managementNumber} (${e.name})";
            }

            return DropdownMenuItem(
              value: e,
              child: Text(_text, style: TextStyle(fontSize: 13)),
            );
          },
        ).toList(),
        onChanged: (e) {
          if (onChanged != null) onChanged(e);
        },
      ),
    ),
  );
}
