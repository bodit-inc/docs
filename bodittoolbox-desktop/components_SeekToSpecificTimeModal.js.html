import React from "react";
import PropTypes from "prop-types";
import InputMask from "react-input-mask";
import { Modal, Button, Form } from "react-bootstrap";
import moment from "moment";
import "moment/locale/ko";

/**
 * 특정 시간대로 탐색하기 팝업 컴포넌트입니다.
 */
class SeekToSpecificTimeModal extends React.Component {
  constructor(props) {
    super(props);

    this.state = {
      targetRecordTime: props.defaultRecordTime,
    };
  }

  componentDidMount() {}

  componentWillUnmount() {}

  /**
   * 탐색 이벤트를 핸들링합니다.
   * @param {*} e
   */
  onSubmit(e) {
    e.preventDefault();

    const { targetRecordTime } = this.state;
    const { onSeekTo } = this.props;

    onSeekTo(moment.duration(targetRecordTime).asSeconds());
  }

  /**
   * 탐색 시간 수정 이벤트를 핸들링합니다.
   * @param {*} e
   */
  onChange(e) {
    this.setState({ targetRecordTime: e.target.value });
  }

  render() {
    const { onHide } = this.props;
    const { targetRecordTime } = this.state;

    return (
      <Modal show={true} size="sm" animation={false} onHide={onHide.bind(this)} centered>
        <Modal.Header closeButton>
          <Modal.Title>특정 시간대로 탐색</Modal.Title>
        </Modal.Header>
        <Form onSubmit={this.onSubmit.bind(this)}>
          <Modal.Body>
            <InputMask mask="99:99:99.999" autoFocus onChange={this.onChange.bind(this)} value={targetRecordTime} className="form-control my-input" />
          </Modal.Body>
          <Modal.Footer>
            <Button type="submit" variant="primary" size="sm">
              탐색
            </Button>
          </Modal.Footer>
        </Form>
      </Modal>
    );
  }
}

SeekToSpecificTimeModal.defaultProps = {
  defaultRecordTime: "00:00:00.000",
};
SeekToSpecificTimeModal.propTypes = {
  defaultRecordTime: PropTypes.string,
  onHide: PropTypes.func.isRequired,
  onSeekTo: PropTypes.func.isRequired,
};

export default SeekToSpecificTimeModal;
