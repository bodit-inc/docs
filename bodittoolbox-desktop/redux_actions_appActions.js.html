import { SHOW_TOAST, SET_APP_STATE, RESET_APP_STATE } from "../constants/actionTypes";

/**
 * Toast 메시지를 보여줍니다.
 * @param {*} message
 * @param {*} duration
 */
export function showToast(message, duration) {
  return (dispatch) => {
    dispatch({
      type: SHOW_TOAST,
      data: {
        message,
        duration: duration ? duration : 1000,
      },
    });
    return Promise.resolve();
  };
}

/**
 * app state를 세팅합니다.
 * @param {*} data
 */
export function setAppState(data) {
  return (dispatch) => {
    return new Promise((resolve, reject) => {
      dispatch({
        type: SET_APP_STATE,
        data: data,
      });
      resolve();
    });
  };
}

/**
 * app state를 리셋합니다.
 */
export function resetAppState() {
  return (dispatch) => {
    return new Promise((resolve, reject) => {
      dispatch({
        type: RESET_APP_STATE,
      });
      resolve();
    });
  };
}
