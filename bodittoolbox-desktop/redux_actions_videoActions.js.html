import { SET_VIDEO_STATE, RESET_VIDEO_STATE, SET_VIDEO_CURRENT_DURATION } from "../constants/actionTypes";

/**
 * 현재 총 재생시간을 가지는 currentDuration 값을 업데이트 합니다.
 * @param {*} data
 * @returns
 */
export function setVideoCurrentDuration(data) {
  return (dispatch) => {
    return new Promise((resolve, reject) => {
      dispatch({
        type: SET_VIDEO_CURRENT_DURATION,
        data: data,
      });
      resolve();
    });
  };
}

/**
 * video state를 세팅합니다.
 * @param {*} data
 * @returns
 */
export function setVideoState(data) {
  return (dispatch) => {
    return new Promise((resolve, reject) => {
      dispatch({
        type: SET_VIDEO_STATE,
        data: data,
      });
      resolve();
    });
  };
}

/**
 * video state를 리셋합니다.
 * @returns
 */
export function resetVideoState() {
  return (dispatch) => {
    return new Promise((resolve, reject) => {
      dispatch({
        type: RESET_VIDEO_STATE,
      });
      resolve();
    });
  };
}
