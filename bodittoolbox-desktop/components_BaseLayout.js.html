import React from "react";
import { Toast, Spinner } from "react-bootstrap";
import { useDispatch, useSelector } from "react-redux";
import { setAppState } from "../redux/actions";

/**
 * 모든 페이지에 공통적으로 들어가는 레이아웃
 */
export const BaseLayout = ({ children }) => {
  const dispatch = useDispatch();
  const app = useSelector((store) => store.app);

  const handleToastClose = () => {
    dispatch(setAppState({ toast: false }));
  };

  return (
    <div>
      {app.loadingNetwork && (
        <div className="loading-network">
          <Spinner animation="border" variant="primary" role="status" />
        </div>
      )}

      <div className="toast-container">
        <Toast className="mt-4 mr-4" onClose={handleToastClose} show={app.toast} delay={app.toast ? app.toast.duration : 1000} autohide>
          <Toast.Header>
            <strong className="mr-auto">BoditToolBox</strong>
          </Toast.Header>
          <Toast.Body style={{ minWidth: 300, minHeight: 50 }}>{app.toast.message}</Toast.Body>
        </Toast>
      </div>
      {children}
    </div>
  );
};
