export default function dataReducer(state = initialState, action) {
  switch (action.type) {
    case SHOW_TOAST:
      return { ...state, toast: { ...action.data } };
    case SET_APP_STATE:
      return { ...state, ...action.data };
    case RESET_APP_STATE:
      let _initialState = _.cloneDeep(initialState);
      return _initialState;
    default:
      return state;
  }
}
