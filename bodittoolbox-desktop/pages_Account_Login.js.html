import React from "react";
import { connect } from "react-redux";
import { withRouter } from "react-router-dom";
import { setAppState, setAuthState, showToast } from "../../redux/actions";
import { Container, Form, Button } from "react-bootstrap";
import logo from "../../assets/logo_dark.webp";
import { ipcRenderer } from "electron";
import { isLogged, logger } from "../../utils/lib";
import { BaseLayout, SaveDataController } from "../../components";
import { IS_DEV } from "../../utils/appConstants";

/**
 * 로그인 페이지 컴포넌트
 */
class Login extends React.PureComponent {
  constructor(props) {
    super(props);

    this.state = {
      inputId: IS_DEV ? "andy.kim@bodit.co" : "",
      inputPasswd: IS_DEV ? "123123" : "",
    };
  }

  componentDidMount() {
    this.setLoading(true);

    // 이벤트 리스너 추가
    ipcRenderer.on("loginResult", this.onLoginResult);
  }

  componentWillUnmount() {
    // 이벤트 리스너 제거
    ipcRenderer.removeListener("loginResult", this.onLoginResult);
  }

  /**
   * 저장된 데이터 불러오기가 완료되었을때 호출됩니다.
   * @param {*} event
   * @param {*} arg
   */
  onInitialized = (event, arg) => {
    this.setLoading(false);

    if (isLogged(arg.auth)) {
      this.setLoading(true);
      ipcRenderer.send("login", { auth: arg.auth });
    }
  };

  /**
   * 로그인 결과를 가져옵니다. 성공시 tagger페이지로 이동하고 실패시 toast 오류 메시지 띄움.
   * @param {*} event
   * @param {Object} arg - 결과
   * @param {Boolean} arg.success - true: 성공, false: 실패
   * @param {String} arg.message - 결과 메시지
   * @param {Object} arg.auth - 인증 데이터
   */
  onLoginResult = async (event, arg) => {
    logger.info(this.constructor.displayName, `onLoginResult`);

    this.setLoading(false);

    if (arg.success) {
      await this.props.setAuthState(arg.auth);
      this.props.history.push("/tagger");
    } else {
      this.props.showToast(arg.message);
    }
  };

  /**
   * 로딩중인지 표시하는 UI를 On/Off 합니다.
   * @param {Boolean} loading
   */
  setLoading = (loading) => {
    this.props.setAppState({ loadingNetwork: loading });
  };

  /**
   * 로그인 정보 제출을 핸들링합니다.
   * @param {*} e
   */
  handleFormSubmit(e) {
    logger.info(this.constructor.displayName, `handleFormSubmit`);
    e.preventDefault();

    const { inputId, inputPasswd } = this.state;

    this.setLoading(true);

    ipcRenderer.send("login", {
      id: inputId,
      passwd: inputPasswd,
    });
  }

  render() {
    const { inputId, inputPasswd } = this.state;
    const { app } = this.props;

    return (
      <BaseLayout>
        <div className="login-container">
          <SaveDataController onInitialized={this.onInitialized.bind(this)} saveSavedDataIntervalEnabled={false} />
          <img src={logo} alt="logo" />
          <Container>
            <Form onSubmit={this.handleFormSubmit.bind(this)}>
              <Form.Group controlId="formBasicEmail">
                <Form.Label>아이디</Form.Label>
                <Form.Control
                  type="text"
                  placeholder="아이디"
                  value={inputId}
                  onChange={(e) => {
                    this.setState({ inputId: e.target.value });
                  }}
                />
              </Form.Group>
              <Form.Group controlId="formBasicPassword">
                <Form.Label>비밀번호</Form.Label>
                <Form.Control
                  type="password"
                  placeholder="비밀번호"
                  value={inputPasswd}
                  onChange={(e) => {
                    this.setState({ inputPasswd: e.target.value });
                  }}
                />
              </Form.Group>
              <Button variant="primary" type="submit" className="w-100" disabled={app.loadingNetwork}>
                {app.loadingNetwork ? "로그인 중..." : "로그인"}
              </Button>
            </Form>
          </Container>
        </div>
      </BaseLayout>
    );
  }
}

export default withRouter(
  connect(
    (state) => ({
      app: state.app,
    }),
    (dispatch) => ({
      showToast: (message, duration) => dispatch(showToast(message, duration)),
      setAuthState: (dataObject) => dispatch(setAuthState(dataObject)),
      setAppState: (dataObject) => dispatch(setAppState(dataObject)),
    }),
  )(Login),
);
