import React from "react";
import { connect } from "react-redux";
import { setVideoState } from "../redux/actions";
import { ButtonToolbar, ButtonGroup, Button, InputGroup, FormControl } from "react-bootstrap";

/**
 * 탐색 간격 조절 컴포넌트
 */
class SeekTimeIntervalController extends React.PureComponent {
  handleChangeSeekTimeInterval = (value) => {
    if (value === 0 && this.lastSeekTimeIntervalSeconds === "-") {
      value = "-0";
    }
    this.lastSeekTimeIntervalSeconds = "" + value;

    this.props.setVideoState({
      videoSeekTimeIntervalSeconds: value,
    });
  };

  render() {
    const { video, visiblePreset } = this.props;
    return (
      <ButtonToolbar className="d-flex w-100">
        <ButtonGroup size="small" className="flex-grow-1">
          <Button
            as="a"
            variant="dark"
            className="rounded-0 border-0 w-100"
            size="sm"
            onClick={(e) => {
              this.handleChangeSeekTimeInterval("1");
            }}
          >
            간격
          </Button>

          {visiblePreset.map((val) => (
            <Button
              key={"x" + val}
              as="a"
              className="rounded-0 border-0 w-100"
              size="sm"
              onClick={(e) => {
                this.handleChangeSeekTimeInterval(val);
              }}
              variant={parseFloat(video.videoSeekTimeIntervalSeconds) === val ? "secondary" : "outline-secondary"}
            >
              {val}
            </Button>
          ))}
        </ButtonGroup>
        <InputGroup size="sm" style={{ width: 80 }}>
          <FormControl
            type="number"
            variant="dark"
            className=""
            value={video.videoSeekTimeIntervalSeconds}
            step="250"
            onChange={(e) => {
              this.handleChangeSeekTimeInterval(e.target.value);
            }}
          />
        </InputGroup>
      </ButtonToolbar>
    );
  }
}

export default connect(
  (state) => ({
    video: state.video,
  }),
  (dispatch) => ({
    setVideoState: (dataObject) => dispatch(setVideoState(dataObject)),
  }),
)(SeekTimeIntervalController);
