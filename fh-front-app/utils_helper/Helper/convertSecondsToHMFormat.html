static String convertSecondsToHMFormat(int seconds) {
  int min = (seconds / 60).roundToDouble().toInt();
  int hour = (min ~/ 60).roundToDouble().toInt();
  int remainMin = min - hour * 60;
  return "${hour.abs()}h ${remainMin.abs()}'";
}
