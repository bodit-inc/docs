onTapDetail() async {
  final String log = "${widget.reportCardViewVM.reportCardTitle} " + "${widget.reportCardViewVM.isDetailOpen ? "접기" : "자세히보기"}" + " 누름";
  LogManager().addLog(log, action: LogConstants.buttonTap, screen: runtimeType.toString());
  widget.reportCardViewVM.toggleDetailOpen();
  setState(() {});

  if (widget.reportCardViewVM.isDetailOpen && widget.reportCardViewVM.report.actions != null) {
    widget.reportCardViewVM.barItemsDay = null;
    await widget.reportCardViewVM.fetchGraphData();
    await scrollToEnd();
  }
}
