String get name {
  switch (this) {
    case GraphViewType.H24:
      return "24H";
    case GraphViewType.H48:
      return "48H";
    case GraphViewType.W1:
      return "1주";
    case GraphViewType.M1:
      return "1달";
    case GraphViewType.M3:
      return "3달";
    default:
      return "1주";
  }
}
