String get imagePath {
  switch (this) {
    case HomeMenuType.healthDoubt:
      return "assets/images/home_menu/health_doubt.png";
    case HomeMenuType.oddAction:
      return "assets/images/home_menu/odd_action.png";
    case HomeMenuType.needCheck:
      return "assets/images/home_menu/need_check.png";
    case HomeMenuType.question:
      return "assets/images/home_menu/chat.png";
    default:
      return "";
  }
}
