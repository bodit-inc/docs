void sortReportsByPinned() {
  /// 날짜 내림차순으로 먼저 정렬합니다.
  _reports!.sort((a, b) {
    if (a.targetAt!.millisecondsSinceEpoch > b.targetAt!.millisecondsSinceEpoch)
      return -1;
    else if (a.targetAt!.millisecondsSinceEpoch < b.targetAt!.millisecondsSinceEpoch)
      return 1;
    else
      return 0;
  });

  /// pinned 리포트와 unPinned리포트를 분리하고 다시 합쳐 pinned 리포트들이 상단으로 오게합니다.
  List<Report> pinnedReports = _reports!.where((element) => element.isPinned).toList();
  List<Report> unPinnedReports = _reports!.where((element) => !element.isPinned).toList();
  _reports = [...pinnedReports, ...unPinnedReports];
}
