Called whenever the object needs to paint. The given Canvas has its coordinate space configured such that the origin is at the top left of the box. The area of the box is the size of the size argument.

Paint operations should remain inside the given area. Graphical operations outside the bounds may be silently ignored, clipped, or not clipped. It may sometimes be difficult to guarantee that a certain operation is inside the bounds (e.g., drawing a rectangle whose size is determined by user inputs). In that case, consider calling Canvas.clipRect at the beginning of paint so everything that follows will be guaranteed to only draw within the clipped area.

Implementations should be wary of correctly pairing any calls to Canvas.save/Canvas.saveLayer and Canvas.restore, otherwise all subsequent painting on this canvas may be affected, with potentially hilarious but confusing results.

To paint text on a Canvas, use a TextPainter.

To paint an image on a Canvas:

  1. Obtain an ImageStream, for example by calling ImageProvider.resolve on an AssetImage or NetworkImage object.

  2. Whenever the ImageStream's underlying ImageInfo object changes (see ImageStream.addListener), create a new instance of your custom paint delegate, giving it the new ImageInfo object.

  3. In your delegate's paint method, call the Canvas.drawImage, Canvas.drawImageRect, or Canvas.drawImageNine methods to paint the ImageInfo.image object, applying the ImageInfo.scale value to obtain the correct rendering size.

@override
void paint(Canvas canvas, Size size) {
  if (barItemsDay == null) return;
  horizontalLinePaint.strokeWidth = horizontalLineHeight;
  yMaxVal = Helper().getMaxValueFromBarDataDay(barDatasDay: barItemsDay!);

  maxBarHeight =
      (size.height - xLabelHourSize - xLabelHourMargin - xLabelDaySize - xLabelDayMargin - xLabelBirthDaySize - xLabelBirthDayMargin - yTopMargin - horizontalLineHeight) / (isBiDirectional ? 2 : 1);
  baseOffsetY = isBiDirectional ? -maxBarHeight : 0;
  ySectionHeight = maxBarHeight / (yLabelCount - 1);
  if (yMaxVal == 0) yMaxVal = 1;
  barHeightRatio = (maxBarHeight / yMaxVal);
  var offsetDx = offset.dx;

  if (!yLabelVisible) {
    barStartOffsetX = 0;
    yLabelSize = 0;
    yLabelMargin = 0;
    yLabelWidth = 0;
  }

  // 낮, 밤 표시
  for (var k = 0; k < barItemsDay!.length; k++) {
    var barDataHour = barItemsDay![k].barDataHour;
    offsetDx = offset.dx + ((barWidth * hourCntInDay + barSpace * hourCntInDay) * k);

    // 밴드 하단
    for (var i = 0; i < barDataHour.length; i++) {
      if (barDataHour[i] == null) continue;
      if ((k == 0 && i == 0 && (barDataHour[i]!.dateTime.hour >= barItemsDay![k].sunSetDateTime.hour || barDataHour[i]!.dateTime.hour < barItemsDay![k].sunRiseDateTime.hour)) ||
          barDataHour[i]!.dateTime.hour == barItemsDay![k].sunSetDateTime.hour) {
        var width = (barWidth + barSpace) * (barItemsDay![k].sunSetDateTime.hour - barItemsDay![k].sunRiseDateTime.hour);
        var _dx = offsetDx +
            yLabelWidth +
            yLabelMargin +
            barStartOffsetX +
            (barWidth + barSpace) * i +
            (barWidth + barSpace) * ((barItemsDay![k].sunSetDateTime.hour - barItemsDay![k].sunRiseDateTime.hour) / 2);
        if (k == 0 && i == 0) {
          if (barDataHour[i]!.dateTime.hour >= barItemsDay![k].sunSetDateTime.hour) {
            _dx = offsetDx + yLabelWidth + yLabelMargin + barStartOffsetX + (barWidth + barSpace) * ((barItemsDay![k].sunSetDateTime.hour).abs() / 2);
            width = (barWidth + barSpace) * (barDataHour[i]!.dateTime.hour - barItemsDay![k].sunSetDateTime.hour).abs();
          } else if (barDataHour[i]!.dateTime.hour < barItemsDay![k].sunRiseDateTime.hour) {
            _dx = offsetDx + yLabelWidth + yLabelMargin + barStartOffsetX + (barWidth + barSpace) * ((barDataHour[i]!.dateTime.hour - barItemsDay![k].sunRiseDateTime.hour).abs() / 2);
            width = (barWidth + barSpace) * (barDataHour[i]!.dateTime.hour - barItemsDay![k].sunRiseDateTime.hour).abs();
          }
        }

        if (isNightVisible) {
          canvas.drawRect(
            Rect.fromCenter(
              center: Offset(
                _dx,
                (size.height - xLabelHourSize - xLabelHourMargin - xLabelDaySize - xLabelDayMargin) / 2,
              ),
              width: width,
              height: size.height - xLabelHourSize - xLabelHourMargin - xLabelDaySize - xLabelDayMargin,
            ),
            nightPaint,
          );
        }
      }
    }
  }

  // x축 일별 순회
  for (var k = 0; k < barItemsDay!.length; k++) {
    Offset barCenter = Offset.zero;
    double barHeight = size.height;
    List<BarDataHour?> barDataHour = barItemsDay![k].barDataHour;

    // if (barDayItems?[k] == null)
    //   barDataHour = [BarDataHour(value: 0, valueLabel: "0", realValue: 0, label: "-", dateTime: DateTime.now())];
    // else
    //   barDataHour = barDayItems![k].barDataHour;

    offsetDx = offset.dx + ((barWidth * hourCntInDay + barSpace * hourCntInDay) * k);
    String labelDay = (hourCntInDay == 1) ? "" : intl.DateFormat('yyyy.MM.dd').format(barItemsDay![k].dateTime);

    // 가로줄 그리기
    if (yLabelVisible || horizontalLineVisible) drawYLabel(canvas, size, k);

    // xLabelDay 그리기
    if (xLabelHourVisible) drawXLabelDay(canvas, size, Offset(offsetDx, 0), barItemsDay![k].label, labelDay, k);

    // x축 - 시간 순회
    for (var i = 0; i < hourCntInDay; i++) {
      if (barDataHour[i] == null) continue;
      barCenter = Offset.zero;
      barHeight = size.height;
      double barHeightBottom = size.height;

      // Bar 그리기
      List<dynamic> result = drawBar(canvas, size, barDataHour, offsetDx, i);
      barCenter = result[0];
      barHeight = result[1];
      barHeightBottom = result[2];

      // xLabelValue 텍스트 그리기
      if (xLabelValueVisible) drawXLabelValue(canvas, size, barDataHour[i]!.valueLabel, barDataHour[i]!.valueBottomLabel, barCenter, barHeight, barHeightBottom);

      // xLabelHour 그리기
      if (xLabelHourVisible) drawXLabelHour(canvas, size, barCenter, barDataHour[i]!.label, labelDay);

      // xLabelBirthDay 그리기
      if (xLabelHourVisible)
        drawXLabelBirthDay(
          canvas,
          size,
          barCenter,
          cow?.getDayAfterBirthStrFromDateTime(barDataHour[i]!.dateTime) ?? "-",
          labelDay,
        );
    }
  }

  // 정상범위 밴드 표시
  if (isRangeVisible) {
    drawRangePath(canvas, size, false);

    if (isBiDirectional) {
      drawRangePath(canvas, size, true);
    }
  }

  // 평균 선 그래프
  if (isAvgLineVisible) {
    drawAvgLine(canvas, size, false);
    if (isBiDirectional) drawAvgLine(canvas, size, true);
  }
}
